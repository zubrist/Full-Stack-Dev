let vs var in JavaScript

In JavaScript, both let and var are used for variable declaration, but they have some key differences:

  • Similarities:
    • Both let and var are used to declare variables in JavaScript.
    • Both can be reassigned values.
  • Differences:
    • var is function-scoped, meaning it is available throughout the function in which it is declared.
    • let is block-scoped, meaning it is only available within the block (enclosed by curly braces) in which it is declared. This makes code more predictable and avoids potential scoping issues.
    • Variables declared with var are hoisted to the top of their function or global scope, which can sometimes lead to unexpected behavior if you use the variable before it's declared. let avoids this issue.
    • Variables declared with let are not hoisted, and trying to access them before they are declared will result in a `ReferenceError`.

Example: Scoping with let vs var vs const

Here's an example to illustrate the difference in scoping between let and var:

function withVar() {
  var x = 10;
  if (true) {
    var x = 20; // Re-declares x within the if block, but still accessible outside
    console.log("Inside if:", x); // Output: 20
  }
  console.log("Outside if:", x); // Output: 20 (x is accessible throughout the function)
}

function withLet() {
  let y = 10;
  if (true) {
    let y = 20; // Creates a new local variable y within the if block
    console.log("Inside if:", y); // Output: 20
  }
  console.log("Outside if:", y); // Output: 10 (y declared with let is only accessible within the block)
}

function withConst() {
    const PI = 3.14159;  // Constant value cannot be reassigned later
    // PI = 3.14;  // This line will cause an error (attempt to reassign const)
    if (true) {
      // const PI = 3.14;  // This line will also cause an error (cannot redeclare const)
      console.log("Inside if:", PI); // Output: 3.14159
    }
    console.log("Outside if:", PI); // Output: 3.14159 (PI remains constant)
  }

withVar();
withLet();
withConst();
